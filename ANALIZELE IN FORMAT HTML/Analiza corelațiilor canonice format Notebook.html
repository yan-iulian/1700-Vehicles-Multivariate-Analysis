def test_bartlett(r2,n,p,q,m):
    v=1-r2
    chi2=(-n+1+(p+q+1)/2)*np.log(np.flip(np.cumprod(np.flip(v))))
    n_lib=[ (p-k+1)*(q-k+1)
        for k in range(1,(m+1))
    ]
    p_values=1-stts.chi2.cdf(chi2,n_lib)
    return p_values
